104 function drawRotatedImage(image, x, y, angle) {
105  // save the current co-ordinate system
106  // before we screw with it
107  context.save();
108  // move to the middle of where we want to draw our image
109  context.translate(x, y);
110  // rotate around that point, converting our
111  // angle from degrees to radians
112  context.rotate(angle * TO_RADIANS);
113  // draw it up and to the left by half the width
114  // and height of the image
115  context.drawImage(image, -(image.width/2), -(image.height/2));
116  // and restore the co-ords to how they were when we began
117  context.restore();
118 }
